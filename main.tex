            \begin{itemize}

                \item \textbf{Approach}: Design of robust tracking loops \cite{barreauKalmanFilterBased2012, psiakiExtendedKalmanFilter2002,florindoMultiFrequencyKalmanFilter2024,fohlmeisterDualKalmanFiltering2018};
                \item \textbf{Trade-off}: Bayesian, but not tailored to monitoring applications;

		\column{0.5\textwidth}
        \textbf{For monitoring}
        \begin{visibleenv}<+(1)->
            \begin{itemize}
                \item \textbf{Goal}: Precisely estimate the scintillation-induced fluctuations;
                \item \textbf{Approach}: Operate in open or semi-open loop and explore the geometry knowledge to disentangle the scintillation effects from the receiver's dynamics \cite{xuSemiOpenLoopGNSS2018,lintyOpenLoopReceiverArchitecture2019,gnssDevelopingIonosphericScintillation2014,jiaoEquatorialScintillationAmplitude2016};
                \item \textbf{Trade-off}: Uses the geometry knowledge, but not Bayesian;
            \end{itemize}
        \end{visibleenv}
	\end{columns}

\begin{frame}[c]
	% \frametitle{Goal}
	\vspace{6pt}

	\begin{alertblock}{Main goal}
        Incorporate the geometry information to the Bayesian framework to provide provide robust tracking performance in closed loop, while also providing accurate scintillation monitoring capabilities.
    \end{alertblock}

    \begin{itemize}
        \item<+(1)-> \textbf{Advantage}: Bayesian approach with external information of the receiver--satellite radiolink dynamics, with potential for solving coupling issues between the receiver's dynamics and the scintillation effects;
        \item<+(1)-> \textbf{Disadvantage}: Offline processing is necessary for best performance, but for scientific applications, it is not problematic;
    \end{itemize}
\end{frame}

\begin{frame}[t]
	\frametitle{Kalman filter-based tracking loops}
    Standard \gls{kf}-\gls{pll} formulation:
	\begin{align}
     \underbrace{\begin{bmatrix}
        \hat{\phi} \left[ k \right] \\
        \hat{\dot{\phi}} \left[ k \right] \\
        \hat{\ddot{f}} \left[ k \right]
    \end{bmatrix}}_{= \mathbf{x} \left[ k \right]} & = \underbrace{\begin{bmatrix}
        1 & 2 \pi T & \pi T^2 \\
        0 & 1 & T \\
        0 & 0 & 1
    \end{bmatrix}}_{= \mathbf{F}_{\mathrm{W}}} \underbrace{\begin{bmatrix}
        \hat{\phi} \left[ k-1 \right] \\
        \hat{\dot{\phi}} \left[ k-1 \right] \\
        \hat{\ddot{f}} \left[ k-1 \right]
    \end{bmatrix}}_{= \mathbf{x} \left[ k-1 \right]} + \mathbf{w} \left[ k \right], \quad \mathbf{w} \left[ k \right] \sim \mathcal{N} \left( \mathbf{0}, \mathbf{Q} \right) \\
    \underbrace{\phi \left[ k \right]}_{= z \left[ k \right]} & = \underbrace{\begin{bmatrix}
        1 & 0 & 0
    \end{bmatrix}}_{= \mathbf{H}_{\mathrm{W}}} \underbrace{\begin{bmatrix}
        \hat{\phi} \left[ k \right] \\
        \hat{\dot{\phi}} \left[ k \right] \\
        \hat{\ddot{f}} \left[ k \right]
    \end{bmatrix}}_{= \mathbf{x} \left[ k \right]} + v \left[ k \right], \quad v \left[ k \right] \sim \mathcal{N} \left( 0, \sigma_{v}^{2} \right)
    \end{align}
\end{frame}

\begin{frame}[t]
	\frametitle{Signal model}
	Baseband carrier phase model:
    \begin{align}
        \phi_{\mathrm{T}} \left( t \right) = \underbrace{2\pi \int_{t_{\mathrm{a}}}^{t} f_{\mathrm{D}} \left( u, n_{\mathrm{c}} \right)\dd{u} + \phi_{\mathrm{LOS}} \left( t_{\mathrm{a}}, n_{\mathrm{c}} \right)}_{\textnormal{Line-of-sight dynamics}} + \underbrace{\phi_{\mathrm{atm}} \left( t, n_{\mathrm{c}} \right)}_{\textnormal{Atmospheric effects}}
    \end{align}
    \begin{visibleenv}<+(1)->
        The LOS phase reads
        \begin{align}
            \phi_{\mathrm{LOS}} \left( t_{\mathrm{a}}, n_{\mathrm{c}} \right) & = - 2\pi f_{\mathrm{c}} \overbrace{\frac{\rho_{\mathrm{s,g}} \left( t_{\mathrm{a}}, n_{\mathrm{c}} \right)}{c_{0}}}^{\textnormal{sat-rx dist. [m]}} = - 2\pi f_{\mathrm{c}} \tau_{\mathrm{LOS}}\left( t_{\mathrm{a}}, n_{\mathrm{c}} \right) \nonumber \\
         & = - 2 \pi f_{\mathrm{c}} \int_{t_{t}}^{t_{\mathrm{a}}} \frac{\partial \tau_{\mathrm{LOS}}\left( u, n_{\mathrm{c}} \right)}{\partial u}\dd{u} = 2 \pi \int_{t_{t}}^{t_{\mathrm{a}}} f_{\mathrm{D}} \left( u, n_{\mathrm{c}} \right) \dd{u}% = - \frac{2 \pi f_{\mathrm{c}} \rho_{\mathrm{s,g}}\left( t_{\mathrm{a}}, n_{\mathrm{c}} \right) }{c_{0}}
        \end{align}
    \end{visibleenv}
\end{frame}


\begin{frame}[t]
	\frametitle{Geometry-aided Kalman filter-based tracking loop}
    Considering an \gls{ar} augmented \gls{ssm}, let \(\mathbf{x} \left[ k \right]= \begin{bmatrix}
        \hat{\phi}_{\mathrm{LOS}} \left[ k \right] &
        \hat{f}_{\mathrm{D}} \left[ k \right] &
        \hat{\dot{f}}_{\mathrm{D}} \left[ k \right] &
        \hat{\phi}_{\textnormal{s}} \left[ k \right] &
        \hat{\phi}_{\textnormal{s}} \left[ k-1 \right] &
        \cdots &
        \hat{\phi}_{\textnormal{s}} \left[ k-p-1 \right] &
    \end{bmatrix}^\top\), then
	\begin{align}
     \mathbf{x} \left[ k \right] & = \mathrm{blkdiag} \left( \mathbf{F}_{\mathrm{W}}, \mathbf{F}_{\mathrm{AR}} \right) \mathbf{x} \left[ k-1 \right] + \mathbf{w} \left[ k \right], \quad \mathbf{w} \left[ k \right] \sim \mathcal{N} \left( \mathbf{0}, \mathbf{Q} \right) \\
    \underbrace{\begin{bmatrix}
        \hat{\phi} \left[ k \right] \\
        \hat{\rho}_{\textnormal{s}, \textnormal{g}} \left[ k \right]
    \end{bmatrix}}_{= \mathbf{z} \left[ k \right]} & = \underbrace{\begin{bmatrix}
        1 & 0 & 0 & 1 & 0 & \cdots & 0 \\
        - \frac{2 \pi f_{\mathrm{c}}}{c_{0}} & 0 & 0 & 0 & 0 & \cdots & 0
    \end{bmatrix}}_{= \mathbf{H}} \mathbf{x} \left[ k \right] + \mathbf{v} \left[ k \right], \quad \mathbf{v} \left[ k \right] \sim \mathcal{N} \left( \mathbf{0}, \mathbf{R} \right)
    \end{align}
\end{frame}

\begin{frame}
    \frametitle{Precise ephemerides}

    Availability of Precise ephemerides:
    \begin{itemize}
        \item \textbf{Ultra-rapid} (predicted half): Available in real-time, with an accuracy of roughly 3 to 10 cm.
        \item \textbf{Ultra-rapid} (observed half): Available with a 3- to 9-hour latency, offering < 5 cm accuracy.
        \item \textbf{Rapid}: Available with a 17-hour latency, offering < 2.5 to 5 cm accuracy.
        \item \textbf{Final}: Available after 12 to 18 days, providing the highest orbital accuracy of approximately 2 to 5 cm.
    \end{itemize}

\end{frame}

\begin{frame}
    \frametitle{Conclusions \& Future works}

    \textbf{Conclusions}
    \begin{itemize}
        \item \gls{kf}-based tracking algorithms struggle to provide accurate estimates of the scintillation-induced fluctuations, especially for strong scintillation events;
        \item The incorporation of external information about the receiver--satellite radiolink dynamics can potentially solve coupling issues;
    \end{itemize}

    \textbf{Future works}: nested loop simulation for
    \begin{itemize}
        \item Kalman filter variants
        \item External information sources (e.g., precise ephemerides, inertial sensors, etc.)
        \item Scintillation scenarios (e.g., weak, moderate, and strong scintillation events)
        \item Scintillation models
    \end{itemize}

\end{frame}
